import random def get_random_color(): return random.choice(['Red', 'Green', 'Violet']) def main(): print("=== Color Prediction Game ===") print("Available colors: Red, Green, Violet") score = 0 rounds = 10 for round_number in range(1, rounds + 1): print(f"\nRound {round_number}") user_guess = input("Predict the next color: ").capitalize() if user_guess not in ['Red', 'Green', 'Violet']: print("Invalid input. Please choose Red, Green, or Violet.") continue actual_color = get_random_color() print(f"Actual color: {actual_color}") if user_guess == actual_color: print("Correct prediction! +1 point") score += 1 else: print("Wrong prediction. No point.") print(f"\nGame Over! Your total score: {score}/{rounds}") if __name__ == "__main__": main()
