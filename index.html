import random # Function to generate color prediction and track trends def generate_color_prediction(last_outcome=None): colors = ['Red', 'Green', 'Violet'] # If there's no previous outcome, randomly choose the first color if last_outcome is None: return random.choice(colors) # Trend-following logic: predict based on the last outcome # In this example, we assume that the next outcome will follow the last if last_outcome == 'Red': prediction = 'Green' # Predict Green after Red (example logic) elif last_outcome == 'Green': prediction = 'Violet' # Predict Violet after Green (example logic) else: prediction = 'Red' # Predict Red after Violet (example logic) return prediction # Function to simulate a game with real-time predictions def color_prediction_game(rounds=20): colors = ['Red', 'Green', 'Violet'] game_outcomes = [] predicted_outcomes = [] last_outcome = None # Initialize the first round with no prior outcome for round_num in range(1, rounds + 1): # Simulate the actual outcome (random) actual_outcome = random.choice(colors) game_outcomes.append(actual_outcome) # Predict the outcome based on last round (or random for the first round) prediction = generate_color_prediction(last_outcome) predicted_outcomes.append(prediction) # Update the last outcome to match the actual result last_outcome = actual_outcome # Display round results print(f"Round {round_num}: Actual Outcome = {actual_outcome}, Predicted = {prediction}") return game_outcomes, predicted_outcomes # Start the game simulation if __name__ == "__main__": rounds = 20 # Set the number of rounds for prediction game_outcomes, predicted_outcomes = color_prediction_game(rounds) print("\nGame Outcomes:", game_outcomes) print("Predicted Outcomes:", predicted_outcomes)
